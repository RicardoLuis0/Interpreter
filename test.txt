int fibonacci_loop(int n) {
	if(n<=0)return 0;
	if(n==1)return 1;
	n-=2;
	int i1=1,i2=1,i3=1;
	for(;n>0;n-=1){
		i3=i1;
		i1+=i2;
		i2=i3;
	}
	return i1;
}

int fibonacci_rec (int n) {
	if (n<=0) {
		return 0;
	} else if (n==1) {
		return 1;
	} else {
		return (fibonacci_rec(n-1)+fibonacci_rec(n-2));
	}
}
