int b00;
int b01;
int b02;
int b10;
int b11;
int b12;
int b20;
int b21;
int b22;

int get(int x,int y){
	if(x==0){
		if(y==0){
			return b00;
		}else if(y==1){
			return b01;
		}else if(y==2){
			return b02;
		}
	}else if(x==1){
		if(y==0){
			return b10;
		}else if(y==1){
			return b11;
		}else if(y==2){
			return b12;
		}
	}else if(x==2){
		if(y==0){
			return b20;
		}else if(y==1){
			return b21;
		}else if(y==2){
			return b22;
		}
	}else{
		puts("invalid position");
	}
}

void set(int x,int y,int val){
	if(x==0){
		if(y==0){
			b00=val;
		}else if(y==1){
			b01=val;
		}else if(y==2){
			b02=val;
		}
	}else if(x==1){
		if(y==0){
			b10=val;
		}else if(y==1){
			b11=val;
		}else if(y==2){
			b12=val;
		}
	}else if(x==2){
		if(y==0){
			b20=val;
		}else if(y==1){
			b21=val;
		}else if(y==2){
			b22=val;
		}
	}
}

int turn;

void render(){
	cls();
	puts("TicTacToe\n---\nPlayer "+itos(turn)+"'s Turn\n+---+\n");
	int i,j;
	for(i=0;i<3;i+=1){
	puts("|");
		for(j=0;j<3;j+=1){
			int val=get(j,i);
			if(val==1){
				puts("O");
			}else if(val==2){
				puts("X");
			}else{
				puts(" ");
			}
		}
		puts("|\n");
	}
	puts("+---+\n");
}

int checkwin(){
	if(b00==b01&&b01==b02){
		return b00;
	}else if(b10==b11&&b11==b12){
		return b10;
	}else if(b20==b21&&b21==b22){
		return b20;
	}else if(b00==b10&&b10==b20){
		return b00;
	}else if(b01==b11&&b11==b21){
		return b01;
	}else if(b02==b12&&b12==b22){
		return b02;
	}else if(b00==b11&&b11==b22){
		return b00;
	}else if(b02==b11&&b11==b20){
		return b02;
	}else{
		int is_tie=1,i,j;
		for(i=0;i<3;i+=1){
			for(j=0;j<3;j+=1){
				if(get(i,j)==0)is_tie=0;
			}
		}
		if(is_tie)return 3;
	}
	return 0;
}

int game() {
	turn=1;
	int last_invalid=0;
	int do_loop=1;
	while(do_loop){
		render();
		if(last_invalid){
			last_invalid=0;
			puts("Invalid move\n");
		}
		puts("X:");
		int x=stoi(getline());
		x-=1;
		puts("Y:");
		int y=stoi(getline());
		y-=1;
		if(y<0||x<0||x>2||y>2){
			last_invalid=1;
		}else if(get(x,y)==0){
			set(x,y,turn);
			int check=checkwin();
			if(check>0){
				do_loop=0;
				render();
				if(check==3){
					puts("Tie!\nPress any key...");
				}else{
					puts("Player "+itos(check)+" wins\nPress any key...");
				}
				getch();
			}
			if(turn==1){
				turn=2;
			}else{
				turn=1;
			}
		}else{
			last_invalid=1;
		}
	}
}

int main(){
	cls();
	puts("Welcome to TicTacToe, Press any key to begin...");
	getch();
	int do_loop=1;
	while(do_loop){
		game();
		puts("Play again?(Y/N)\n");
		string out=getline();
		if(out!="Y"&&out!="y"){
			do_loop=0;
		}
	}
	return 0;
}
